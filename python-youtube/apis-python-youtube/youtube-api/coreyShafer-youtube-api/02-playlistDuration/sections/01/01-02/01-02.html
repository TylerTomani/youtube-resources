import os
from googleapiclient.discovery import build
import json
# Get new api key this one is fake
api_key = 'AIzaSyBycc_eAr1_P3NWDzR08IHkCA3_44wKEDM'
youtube = build('youtube','v3',developerKey=api_key)

channel_response  = youtube.channels().list(
    part='contentDetails',
    forUsername='schafer5'
)
response = channel_response.execute()
channel_id = response['items'][0]['id']

playlist_response  = youtube.playlists().list(
    part="contentDetails,snippet",
    channelId=channel_id
)
response = playlist_response.execute()
playlist_id = ''
for pl in response['items']:
    if pl['snippet']['title'] == 'Pandas Tutorials':
        playlist_id = pl['id']
os.system('clear')
print(playlist_id)

pListItems_response = youtube.playlistItems().list(
    part='contentDetails',
    playlistId=playlist_id
)
response = pListItems_response.execute()
os.system('clear')

vid_ids = []
for item in response['items']:
    vid_ids.append(item['id'])

    print(vidId)
